Program                     ::= Declaration*

Declaration                 ::= Statement | FunctionDefinition | StructDefinition
                            | EnumDefinition

Statement                   ::= Expression | VariableDeclaration | FunctionDefinition
                            | IfStatement | ForInLoop | WhileLoop

Expression                  ::= Operation | LambdaFunction

StructDefinition            ::= 'struct' Name StructMember* 'endstruct'
StructMember                ::= VariableDeclaration | FunctionDefinition

EnumDefinition              ::= 'enum' Name+ 'endenum'

FunctionDefinition          ::= 'def' Name '(' ParamList ')' '->' Name FunctionBody 'enddef'
FunctionBody                ::= Statement*

IfStatement                 ::= 'if' Expression 'then' IfClause ElseIfClause* ElseClause? 'endif'

ForInLoop                   ::= 'for' Name 'in' Name 'do' Statement* 'done'
WhileLoop                   ::= 'while' Expression 'do' Statement* 'done'

LambdaFunction              ::= '(' ParamList ')' '->' Expr
